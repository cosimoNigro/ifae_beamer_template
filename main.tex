% headline
\setbeamertemplate{headline}{%
	\begin{beamercolorbox}[sep=1.5ex, wd=\paperwidth, leftskip=0.1cm, rightskip=0cm]{section in head/foot}
	\end{beamercolorbox}%
}
% footline
\setbeamertemplate{footline}{
	\leavevmode%
  	\hbox{%
  		\begin{beamercolorbox}[wd=.4\paperwidth, ht=2.25ex, dp=1ex,center]{author in head/foot}%
    		\usebeamerfont{author in head/foot}\insertshortauthor
  		\end{beamercolorbox}%
		\begin{beamercolorbox}[wd=.6\paperwidth, ht=2.25ex, dp=1ex,center]{title in head/foot}%
    		\usebeamerfont{title in head/foot}%\insertshorttitle\hspace*{3em}
 		\insertframenumber{} / \inserttotalframenumber\hspace*{1ex}
  		\end{beamercolorbox}
  	}%
	\vskip0pt%
}
	% get rid of the navigation symbols
	\beamertemplatenavigationsymbolsempty
	% set the bullet for itemize 
\setbeamertemplate{itemize item}{
	\normalsize\raise 1.5 pt\hbox{
		\donotcoloroutermaths{\color{ifaeorange}$\bm{>}\mkern-18mu$}
	}
}
\setbeamertemplate{itemize subitem}{
	\normalsize\raise 1.5 pt\hbox{
		\donotcoloroutermaths{\color{ifaeorange}$\bm{\rightarrow}\mkern-18mu$}
	}
}
